 package com.digitalmoneyhouse.accountservice.util;
 
 import org.springframework.stereotype.Component;
 
 import java.util.Collections;
 import java.util.List;
 
 @Component
 public class AliasGenerator {
 
     private final List<String> baseWords = List.of(
             "luna", "sol", "rio", "verde", "nube",
             "pampa", "campo", "fuego", "mar", "sur"
     );
 
     public String generate() {
         List<String> words = new java.util.ArrayList<>(baseWords);
         Collections.shuffle(words);
         return words.get(0) + "." + words.get(1) + "." + words.get(2);
     }
 }
